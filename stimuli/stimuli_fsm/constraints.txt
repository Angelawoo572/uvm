// FSM Constraints Examples

// 1. inside - Bounded LFSR only
constraint force_rst { rst_n == 1'b0 };
constraint test_addr1 { addr inside {[11:20]} };

// 2. addr > a, addr < b - Bounded LFSR with range mapping
constraint test_addr_range1 { addr > 4'h4 && addr < 4'hC };
constraint test_addr_range2 { addr_bus > 8'h20 && addr_bus < 8'h80 };

// 3. enum types - Bounded LFSR with enum constraints
constraint state_enum { state inside {IDLE, FETCH, EXECUTE, WRITE} };

// 4. dist - Bounded LFSR with weight table
constraint test_payload { payload dist { 0 := 5, 1 := 95 } };
constraint test_data { data dist { 0 := 10, 20 := 80, 12 := 10 } };

// 5. bit-wise constraint (parity) - Bounded LFSR with bit masking
constraint test_parity { parity == (^data) };
constraint test_lsb { data[0] == 1'b1 };
constraint test_msb_range { data[7:4] != 4'b1111 };
